import theano
def LearningAsReachedALayer(SampleFaceImage,UntilWhichLayer):
    colormaps=True
    feedForwardOnCNN= theano.function([CNNmodelForFaceExpresions.layers[0].input],CNNmodelForFaceExpresions.layers[UntilWhichLayer].output, allow_input_downcast=True)
    
    ForwardPropResultAtRequiredlayer = feedForwardOnCNN(SampleFaceImage)
    ## Now each unit has learned something and lets iterate over the units of the result and plot. 
    #The imge is changed as it reached various layers
    Totalfilters = ForwardPropResultAtRequiredlayer.shape[1]
    
    fig = plt.figure(figsize=(15,10))
    
    for EachUnit in range(Totalfilters):
        SubPlotAreaProps = fig.add_subplot(Totalfilters/16,16,EachUnit+1)
        SubPlotAreaProps.imshow(ForwardPropResultAtRequiredlayer[0,EachUnit,:,:], cmap='Blues')        
        ## below are common to avoid the axes nubmbering eetc.. while plotting
        plt.xticks(np.array([]))
        plt.yticks(np.array([]))
        plt.tight_layout()
    plt.show()
